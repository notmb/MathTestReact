rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Public read for one-time test content.
    match /Subjects/Math/Algebra/Topics/Mix/{docId=**} {
      allow read: if true;
    }
    match /Subjects/Math/Algebra/Topics/Retaking/{docId=**} {
      allow read: if true;
    }

    // One-time links and finalization status.
    match /Subjects/Math/TestLinks/{id} {
      allow read: if true;
      allow create, update: if true;
    }

    // Answers/result per one-time link.
    match /Subjects/Math/TestLinks/{linkId}/testResults/{docId} {
      allow read, create, update: if true;
    }

    // Student summary scores.
    match /Subjects/Math/MyStudents/{studentId} {
      allow read: if true;

      // Create only score maps (if student profile doc is absent).
      allow create: if
        request.resource.data.keys().hasOnly([
          "testScores",
          "testScoresRetaking"
        ]);

      // Allow only score map changes on existing profile.
      allow update: if
        request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          "testScores",
          "testScoresRetaking"
        ]);
    }

    // Detailed result for a student + variant.
    match /Subjects/Math/MyStudents/{studentId}/ResultsTest/{testId} {
      allow read: if true;

      allow create: if
        request.resource.data.keys().hasOnly([
          "pointsForTasks",
          "userAnswers",
          "result",
          "variantId",
          "variantName"
        ]);

      // Allow updating only result fields; keep variantId immutable.
      allow update: if
        request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          "pointsForTasks",
          "userAnswers",
          "result",
          "variantName"
        ]) &&
        request.resource.data.variantId == resource.data.variantId;
    }

    // Everything else is authenticated only.
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
